120 static Document create(const std::filesystem::path& inputPath)
121 {
122 std::ifstream jsonFile;
123 jsonFile.exceptions(std::ios::failbit | std::ios::badbit);
124 jsonFile.open(inputPath);
125 if (!jsonFile.is_open()) {
126 throw std::runtime_error("Unable to open JSON file: " + inputPath.u8string());
127 }
128 return Document(jsonFile);
129 }
130
137 void save(const std::filesystem::path& outputPath, std::optional<int> indentSpaces) const
138 {
139 std::ofstream file;
140 file.exceptions(std::ofstream::failbit | std::ofstream::badbit);
141 file.open(outputPath, std::ios::out | std::ios::binary);
142 if (!file.is_open()) {
143 throw std::runtime_error("Unable to write to JSON file: " + outputPath.u8string());
144 }
145 file << root()->dump(indentSpaces.value_or(-1));
146 file.close();
147 }

154 std::optional<std::string> validate(const std::optional<std::string>& jsonSchema = std::nullopt) const;
