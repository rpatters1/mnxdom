87public:

104 Document(std::istream& inputStream) : Object(std::make_shared<json>(json::object()), json_pointer{})
105 {
106 inputStream >> *root();
107 }
108
109 MNX_REQUIRED_CHILD(Global, global);
110 MNX_REQUIRED_CHILD(MnxMetaData, mnx);
111 MNX_OPTIONAL_CHILD(Array<Layout>, layouts);
112 MNX_REQUIRED_CHILD(Array<Part>, parts);
113 MNX_OPTIONAL_CHILD(Array<Score>, scores);
114
121 static Document create(const std::filesystem::path& inputPath)
122 {
123 std::ifstream jsonFile;
124 jsonFile.exceptions(std::ios::failbit | std::ios::badbit);
125 jsonFile.open(inputPath);
126 if (!jsonFile.is_open()) {
127 throw std::runtime_error("Unable to open JSON file: " + inputPath.u8string());
128 }
129 return Document(jsonFile);
130 }
131
138 void save(const std::filesystem::path& outputPath, std::optional<int> indentSpaces) const
139 {
140 std::ofstream file;
141 file.exceptions(std::ofstream::failbit | std::ofstream::badbit);
142 file.open(outputPath, std::ios::out | std::ios::binary);
143 if (!file.is_open()) {
144 throw std::runtime_error("Unable to write to JSON file: " + outputPath.u8string());
145 }
146 file << root()->dump(indentSpaces.value_or(-1));
147 file.close();
148 }
149
154 template <typename T>
155 T get(const json_pointer& jsonPointer)
156 {
157 static_assert(std::is_base_of_v<Base, T>, "template class must be derived from mnx::Base.");
158 return T(root(), jsonPointer);
159 }
160
163 template <typename T>
164 T get(const std::string& jsonPointerString)
165 { return get<T>(json_pointer(jsonPointerString)); }
166
167private:
168 friend validation::ValidationResult validation::schemaValidate(const Document& document, const std::optional<std::string>& jsonSchema);
169};
