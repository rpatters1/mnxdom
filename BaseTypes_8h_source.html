97struct ArrayAppendFromMake<T, MakeFunc>
98{
103 T append(Args... args)
104 {
105 return static_cast<Array<T>&>(*this).template appendImpl<T>(std::forward<Args>(args)...);
106 }

110template <typename Derived, typename T>
111struct ArrayAppendBase
112{
117 template <typename U = T, typename... Args,
118 std::enable_if_t<std::is_base_of_v<Base, U>, int> = 0>
119 U append(Args&&... args)
120 {
121 return static_cast<Derived&>(*this).template appendImpl<U>(std::forward<Args>(args)...);
122 }
123};

125template <typename Derived, typename T, typename = void>
126struct ArrayAppendOverloads : ArrayAppendBase<Derived, T> {};
127
128template <typename Derived, typename T>
129struct ArrayAppendOverloads<Derived, T, std::void_t<decltype(&T::make)>>
130 : ArrayAppendBase<Derived, T>,
131 ArrayAppendFromMake<T, &T::make>
132{
133 using ArrayAppendBase<Derived, T>::append;
134 using ArrayAppendFromMake<T, &T::make>::append;
135};

189class Base
190{
191public:
192 virtual ~Base() = default;
193
195 Base(const Base& src) : m_root(src.m_root), m_pointer(src.m_pointer)
196 {}
197
199 Base(Base&& src) noexcept : m_root(src.m_root), // m_root must be copied (not moved)
200 m_pointer(std::move(src.m_pointer))
201 {}
202
204 Base& operator=(const Base& src)
205 {
206 if (this != &src) {
207 if (m_root != src.m_root) {
208 throw std::logic_error("Assignment from a different JSON document is not allowed.");
209 }
210 m_pointer = src.m_pointer;
211 }
212 return *this;
213 }
214
216 Base& operator=(Base&& src)
217 {
218 if (this != &src) {
219 if (m_root != src.m_root) {
220 throw std::logic_error("Assignment from a different JSON document is not allowed.");
221 }
222 m_pointer = std::move(src.m_pointer);
223 }
224 return *this;
225 }
226
229 [[nodiscard]] std::string dump(int indents = -1) const
230 {
231 return ref().dump(indents);
232 }
233
237 template <typename T>
238 [[nodiscard]] T parent() const
239 {
240 static_assert(std::is_base_of_v<Base, T>, "Template type mush be derived from Base.");
241 return T(m_root, m_pointer.parent_pointer());
242 }
243
248 template <typename T, typename Scope = scope::Default>
249 [[nodiscard]] std::optional<T> getEnclosingElement() const;
250
252 [[nodiscard]] json_pointer pointer() const { return m_pointer; }
253
255 [[nodiscard]] Document document() const;
256
257protected:
263 [[nodiscard]] json& ref() const { return resolve_pointer(); }
264
269 [[nodiscard]] json& ref() { return resolve_pointer(); }
270
272 [[nodiscard]] const std::shared_ptr<json>& root() const { return m_root; }
273
279 Base(const std::shared_ptr<json>& root, json_pointer pointer)
280 : m_root(root), m_pointer(std::move(pointer)) {}
281
288 Base(json&& jsonRef, Base& parent, std::string_view key)
289 : m_root(parent.m_root), m_pointer(parent.m_pointer / std::string(key))
290 {
291 (*m_root)[m_pointer] = std::move(jsonRef);
292 }
293
301 template <typename T>
302 [[nodiscard]] T getChild(std::string_view key) const
303 {
304 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
305
306 json_pointer childPointer = m_pointer / std::string(key);
307 if (!checkKeyIsValid<T>(childPointer)) {
308 throw std::out_of_range("Child node not found: " + std::string(key));
309 }
310
311 return T(m_root, childPointer);
312 }
313
321 template <typename T>
322 T setChild(std::string_view key, const T& value)
323 {
324 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
325
326 json_pointer childPointer = m_pointer / std::string(key);
327 (*m_root)[childPointer] = value.ref();
328 return T(m_root, childPointer);
329 }
330
338 template <typename T>
339 [[nodiscard]] std::optional<T> getOptionalChild(std::string_view key) const
340 {
341 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
342
343 json_pointer childPointer = m_pointer / std::string(key);
344 if (!checkKeyIsValid<T>(childPointer)) {
345 return std::nullopt;
346 }
347
348 return T(m_root, childPointer);
349 }
350
351private:
359 template <typename T>
360 [[nodiscard]] bool checkKeyIsValid(const json_pointer& pointer) const
361 {
362 if (!(*m_root).contains(pointer)) {
363 return false;
364 }
365
366 const json& node = (*m_root).at(pointer);
367
368 if constexpr (std::is_base_of_v<Object, T>) {
369 if (!node.is_object()) {
370 throw std::runtime_error("Expected an object for: " + pointer.to_string());
371 }
372 } else if constexpr (std::is_base_of_v<Array<typename T::value_type>, T>) {
373 if (!node.is_array()) {
374 throw std::runtime_error("Expected an array for: " + pointer.to_string());
375 }
376 }
377
378 return true;
379 }
380
386 [[nodiscard]] json& resolve_pointer() const
387 {
388 return (*m_root).at(m_pointer); // Throws if invalid
389 }
390
391 const std::shared_ptr<json> m_root;
392 json_pointer m_pointer;
393
394 friend class validation::SemanticValidator;
395};

469template <typename T>
470class Array : public Base,
471 public detail::ArrayAppendOverloads<Array<T>, T>
472{
473 static_assert(std::is_arithmetic_v<T> || std::is_same_v<T, std::string> ||
474 std::is_base_of_v<ArrayElementObject, T>, "Invalid MNX array element type.");
475
476private:
477 template<typename ArrayType>
478 struct iter
479 {
480 private:
481 ArrayType* m_ptr;
482 mutable size_t m_idx;
483
484 public:
485 using iterator_category = std::input_iterator_tag;
486 using value_type = T;
487 using difference_type = std::ptrdiff_t;
488 using pointer = void; // not meaningful for proxy/value-returning iterators
489 using reference = T; // since operator* returns by value
490
491 iter(ArrayType* ptr, size_t idx) : m_ptr(ptr), m_idx(idx) {}
492 T operator*() const { return (*m_ptr)[m_idx]; }
493 iter& operator++() { ++m_idx; return *this; }
494 bool operator!=(const iter& o) const { return m_idx != o.m_idx; }
495 };
496
497public:
499 using value_type = T;
500
501 using iterator = iter<Array>;
502 using const_iterator = iter<const Array>;
503
507 Array(const std::shared_ptr<json>& root, json_pointer pointer) : Base(root, pointer)
508 {
509 if (!ref().is_array()) {
510 throw std::invalid_argument("mnx::Array must wrap a JSON array.");
511 }
512 }
513
517 Array(Base& parent, std::string_view key)
518 : Base(json::array(), parent, key) {}
519
521 [[nodiscard]] size_t size() const { return ref().size(); }
522
524 [[nodiscard]] bool empty() const { return ref().empty(); }
525
527 void clear() { ref().clear(); }
528
530 [[nodiscard]] T at(size_t index) const
531 {
532 return operator[](index);
533 }
534
538 [[nodiscard]] T front() const { return (*this)[0]; }
539
543 [[nodiscard]] T back() const { return (*this)[size() - 1]; }
544
546 [[nodiscard]] auto operator[](size_t index) const
547 {
548 checkIndex(index);
549 if constexpr (std::is_base_of_v<Base, T>) {
550 return getChild<T>(std::to_string(index));
551 } else {
552 return getChild<SimpleType<T>>(std::to_string(index));
553 }
554 }
555
557 [[nodiscard]] auto operator[](size_t index)
558 {
559 checkIndex(index);
560 if constexpr (std::is_base_of_v<Base, T>) {
561 return getChild<T>(std::to_string(index));
562 } else {
563 return getChild<SimpleType<T>>(std::to_string(index));
564 }
565 }
566
568 template <typename U = T>
569 std::enable_if_t<!std::is_base_of_v<Base, U>, void>
570 push_back(const U& value)
571 {
572 ref().push_back(value);
573 }
574
576 void erase(size_t index)
577 {
578 checkIndex(index);
579 ref().erase(ref().begin() + index);
580 }
581
586 template <typename U = T>
587 std::enable_if_t<!std::is_base_of_v<Base, U>, std::vector<U>>
588 toStdVector() const
589 { return std::vector<U>(begin(), end()); }
590
592 [[nodiscard]] auto begin() { return iterator(this, 0); }
593
595 [[nodiscard]] auto end() { return iterator(this, size()); }
596
598 [[nodiscard]] auto begin() const { return const_iterator(this, 0); }
599
601 [[nodiscard]] auto end() const { return const_iterator(this, size()); }
602
603protected:
606 void checkIndex(size_t index) const
607 {
608 assert(index < ref().size());
609 if (index >= ref().size()) {
610 throw std::out_of_range("Index out of range");
611 }
612 }
613
614private:
615 template <typename U, typename... Args>
616 U appendImpl(Args&&... args)
617 {
618 static_assert(std::is_base_of_v<Base, U>, "Array::appendImpl requires a Base-derived element type.");
619 if constexpr (std::is_base_of_v<Object, U>) {
620 ref().push_back(json::object());
621 } else {
622 ref().push_back(json::array());
623 }
624 return U(*this, std::to_string(ref().size() - 1), std::forward<Args>(args)...);
625 }

628 friend struct detail::ArrayAppendFromMake;
629 template <typename, typename>
630 friend struct detail::ArrayAppendBase;

689template <typename ContainerType>
690inline ContainerType ArrayElementObject::container() const
691{
692 if constexpr (std::is_base_of_v<ContainerType, ContentObject>) {
693 const auto obj = parent<Array<ArrayElementObject>>().template parent<ContentObject>();

698 throw std::invalid_argument(

700 }
701 return obj.template get<ContainerType>();
702 }
703 } else {
704 return parent<Array<ArrayElementObject>>().template parent<ContainerType>();

803template <typename E, typename = std::enable_if_t<std::is_enum_v<E>>>
804struct EnumStringMapping
805{
806 static const std::unordered_map<std::string, E> stringToEnum();
807
809 static const std::unordered_map<E, std::string> enumToString()
810 {
811 static const std::unordered_map<E, std::string> reverseMap = []() {
812 std::unordered_map<E, std::string> result;
813 for (const auto& element : EnumStringMapping<E>::stringToEnum()) {
814 result.emplace(element.second, element.first);
815 }
816 return result;
817 }();
818 return reverseMap;
819 }

825template <typename T>
826class Dictionary : public Object,
827 public detail::DictionaryAppendOverloads<Dictionary<T>, T>
828{
829 static_assert(std::is_arithmetic_v<T> || std::is_same_v<T, std::string> ||
830 std::is_base_of_v<ArrayElementObject, T>, "Invalid MNX dictionary element type.");
831
832private:
833 template <typename DictionaryType, typename IteratorType>
834 struct iter
835 {
836 using value_type = std::pair<const std::string, T>;
837 using difference_type = std::ptrdiff_t;
838 using iterator_category = std::forward_iterator_tag;
839 using pointer = value_type*;
840 using reference = value_type&;
841
842 private:
843 DictionaryType* m_ptr {};
844 IteratorType m_it {};
845 mutable std::optional<value_type> m_pair; // cached key/value
846
847 void update_pair() const
848 {
849 m_pair.reset();
850 if (m_it != m_ptr->ref().end()) {
851 m_pair.emplace(m_it.key(), m_ptr->valueForKey(m_it.key()));
852 }
853 }
854
855 public:
856 iter(DictionaryType* ptr, IteratorType it)
857 : m_ptr(ptr), m_it(it)
858 {
859 update_pair();
860 }
861
862 [[nodiscard]] reference operator*() const { return *m_pair; }
863 [[nodiscard]] pointer operator->() const { return &*m_pair; }
864
865 iter& operator++() { ++m_it; update_pair(); return *this; }
866 iter operator++(int) { iter tmp(*this); ++(*this); return tmp; }
867
868 [[nodiscard]] bool operator!=(const iter& o) const { return m_it != o.m_it; }
869 [[nodiscard]] bool operator==(const iter& o) const { return m_it == o.m_it; }
870 };
871
872public:
874 using value_type = T;
875
876 using iterator = iter<Dictionary, json::iterator>;
877 using const_iterator = iter<const Dictionary, json::const_iterator>;
878

974 } else {

976 }

978 }
