41[[nodiscard]] inline std::string makeUtcTimestamp()
42{
43 const auto now = std::chrono::system_clock::now();
44 const std::time_t nowTime = std::chrono::system_clock::to_time_t(now);
45 std::tm utcTime {};
46#if defined(_WIN32)
47 gmtime_s(&utcTime, &nowTime);
48#else
49 gmtime_r(&nowTime, &utcTime);
50#endif
51
52 std::ostringstream out;
53 out << std::put_time(&utcTime, "%Y-%m-%dT%H:%M:%SZ");
54 return out.str();
55}

124 MnxdomExtensionData(Base& parent, std::string_view key)
125 : Object(parent, key)
126 {
127 set_provenanceSchemaVersion(MNXDOM_PROVENANCE_SCHEMA_VERSION);
128 set_mnxSchema(getMnxSchemaId());

131 generator.set_version(MNXDOM_VERSION);

138 MNX_OPTIONAL_CHILD(MnxdomClientInfo, client);
139 MNX_OPTIONAL_CHILD(MnxdomSourceInfo, source);
140 MNX_REQUIRED_PROPERTY(std::string, createdAt);
