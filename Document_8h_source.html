150 [[nodiscard]] static Document create(const std::filesystem::path& inputPath)
151 {
152 std::ifstream jsonFile;
153 jsonFile.exceptions(std::ios::failbit | std::ios::badbit);
154 jsonFile.open(inputPath);
155 if (!jsonFile.is_open()) {
156 throw std::runtime_error("Unable to open JSON file: " + inputPath.string());
157 }
158 return Document(jsonFile);
159 }
160
170 template <typename Byte,
171 typename = std::enable_if_t<
172 std::is_integral_v<Byte> &&
173 (sizeof(Byte) == 1) &&
174 !std::is_same_v<std::remove_cv_t<Byte>, bool>>>
175 [[nodiscard]] static Document create(const Byte* data, std::size_t size)
176 {
177 if (!data && size != 0) {
178 throw std::invalid_argument("JSON buffer pointer is null.");
179 }
180 auto root = std::make_shared<json>(nlohmann::json::parse(data, data + size));
181 return Document(root);
182 }
183
190 void save(const std::filesystem::path& outputPath, std::optional<int> indentSpaces) const
191 {
192 std::ofstream file;
193 file.exceptions(std::ofstream::failbit | std::ofstream::badbit);
194 file.open(outputPath, std::ios::out | std::ios::binary);
195 if (!file.is_open()) {
196 throw std::runtime_error("Unable to write to JSON file: " + outputPath.string());
197 }
198 file << root()->dump(indentSpaces.value_or(-1));
199 file.close();
200 }
