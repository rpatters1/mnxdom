88class Base
89{
90public:
91 virtual ~Base() = default;
92
93 // Copy constructor
94 Base(const Base& src) : m_root(src.m_root), m_pointer(src.m_pointer)
95 {}
96
97 // Move constructor
98 Base(Base&& src) noexcept : m_root(src.m_root), // m_root must be copied (not moved)
99 m_pointer(std::move(src.m_pointer))
100 {}
101
102 // Copy assignment operator
103 Base& operator=(const Base& src)
104 {
105 if (this != &src) {
106 if (m_root != src.m_root) {
107 throw std::logic_error("Assignment from a different JSON document is not allowed.");
108 }
109 m_pointer = src.m_pointer;
110 }
111 return *this;
112 }
113
114 // Move assignment operator
115 Base& operator=(Base&& src)
116 {
117 if (this != &src) {
118 if (m_root != src.m_root) {
119 throw std::logic_error("Assignment from a different JSON document is not allowed.");
120 }
121 m_pointer = std::move(src.m_pointer);
122 }
123 return *this;
124 }
125
128 std::string dump(int indents = -1)
129 {
130 return ref().dump(indents);
131 }
132
133protected:
139 json& ref() const { return resolve_pointer(); }
140
145 json& ref() { return resolve_pointer(); }
146
148 const std::shared_ptr<json>& root() const { return m_root; }
149
151 json_pointer pointer() const { return m_pointer; }
152
156 template <typename T>
157 T parent() const
158 {
159 static_assert(std::is_base_of_v<Base, T>, "Template type mush be derived from Base.");
160 return T(m_root, m_pointer.parent_pointer());
161 }
162
168 Base(const std::shared_ptr<json>& root, json_pointer pointer)
169 : m_root(root), m_pointer(std::move(pointer)) {}
170
177 Base(json&& jsonRef, Base& parent, const std::string_view& key)
178 : m_root(parent.m_root), m_pointer(parent.m_pointer / std::string(key))
179 {
180 (*m_root)[m_pointer] = std::move(jsonRef);
181 }
182
190 template <typename T>
191 T getChild(const std::string_view& key) const
192 {
193 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
194
195 json_pointer childPointer = m_pointer / std::string(key);
196 if (!checkKeyIsValid<T>(childPointer)) {
197 throw std::runtime_error("Missing required child node: " + std::string(key));
198 }
199
200 return T(m_root, childPointer);
201 }
202
210 template <typename T>
211 T setChild(const std::string_view& key, const T& value)
212 {
213 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
214
215 json_pointer childPointer = m_pointer / std::string(key);
216 (*m_root)[childPointer] = value.ref();
217 return T(m_root, childPointer);
218 }
219
227 template <typename T>
228 std::optional<T> getOptionalChild(const std::string_view& key) const
229 {
230 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
231
232 json_pointer childPointer = m_pointer / std::string(key);
233 if (!checkKeyIsValid<T>(childPointer)) {
234 return std::nullopt;
235 }
236
237 return T(m_root, childPointer);
238 }
239
240private:
248 template <typename T>
249 bool checkKeyIsValid(const json_pointer& pointer) const
250 {
251 if (!(*m_root).contains(pointer)) {
252 return false;
253 }
254
255 const json& node = (*m_root).at(pointer);
256
257 if constexpr (std::is_base_of_v<Object, T>) {
258 if (!node.is_object()) {
259 throw std::runtime_error("Expected an object for: " + pointer.to_string());
260 }
261 } else if constexpr (std::is_base_of_v<Array<typename T::value_type>, T>) {
262 if (!node.is_array()) {
263 throw std::runtime_error("Expected an array for: " + pointer.to_string());
264 }
265 }
266
267 return true;
268 }
269
275 json& resolve_pointer() const
276 {
277 return (*m_root).at(m_pointer); // Throws if invalid
278 }
279
280 const std::shared_ptr<json> m_root;
281 json_pointer m_pointer;
282};

325template <typename T>
326class Array : public Base
327{
328 static_assert(std::is_same_v<T, int> || std::is_same_v<T, double> ||
329 std::is_same_v<T, bool> || std::is_same_v<T, std::string> ||
330 std::is_base_of_v<ArrayElementObject, T>, "Invalid MNX array element type.");
331
332private:
333 template<typename A>
334 struct iter
335 {
336 A* a;
337 mutable size_t i;
338
339 iter(A* a_, size_t i_) : a(a_), i(i_) {}
340 T operator*() const { return (*a)[i]; }
341 iter& operator++() { ++i; return *this; }
342 bool operator!=(const iter& o) const { return i != o.i; }
343 };
344
345public:
347 using value_type = T;
348
349 using iterator = iter<Array>;
350 using const_iterator = iter<const Array>;
351
355 Array(const std::shared_ptr<json>& root, json_pointer pointer) : Base(root, pointer)
356 {
357 if (!ref().is_array()) {
358 throw std::invalid_argument("mnx::Array must wrap a JSON array.");
359 }
360 }
361
365 Array(Base& parent, const std::string_view& key)
366 : Base(json::array(), parent, key) {}
367
369 size_t size() const { return ref().size(); }
370
372 bool empty() const { return ref().empty(); }
373
375 void clear() { ref().clear(); }
376
378 T operator[](size_t index) const
379 {
380 checkIndex(index);
381 if constexpr (std::is_base_of_v<Base, T>) {
382 return getChild<T>(std::to_string(index));
383 } else {
384 return ref().at(index).template get<T>();
385 }
386 }
387
389 auto operator[](size_t index)
390 {
391 checkIndex(index);
392 if constexpr (std::is_base_of_v<Base, T>) {
393 return getChild<T>(std::to_string(index));
394 } else {
395 return ref().at(index).template get_ref<T&>();
396 }
397 }
398
400 template <typename U = T>
401 std::enable_if_t<!std::is_base_of_v<Base, U>, void>
402 push_back(const U& value)
403 {
404 ref().push_back(value);
405 }
406
411 template <typename U = T, std::enable_if_t<std::is_base_of_v<Base, U>, int> = 0>
412 U append()
413 {
414 if constexpr (std::is_base_of_v<Object, U>) {
415 ref().push_back(json::object());
416 } else {
417 ref().push_back(json::array());
418 }
419 return U(*this, std::to_string(ref().size() - 1));
420 }
421
423 void erase(size_t index)
424 {
425 checkIndex(index);
426 ref().erase(ref().begin() + index);
427 }
428
430 auto begin() { return iterator(this, 0); }
431
433 auto end() { return iterator(this, size()); }
434
436 auto begin() const { return const_iterator(this, 0); }
437
439 auto end() const { return const_iterator(this, size()); }
440
441protected:
444 void checkIndex(size_t index) const
445 {
446 assert(index < ref().size());
447 if (index >= ref().size()) {
448 throw std::out_of_range("Index out of range");
449 }
450 }

552template <typename E, typename = std::enable_if_t<std::is_enum_v<E>>>
553struct EnumStringMapping
554{
555 static const std::unordered_map<std::string, E> stringToEnum();
556
558 static const std::unordered_map<E, std::string> enumToString()
559 {
560 static const std::unordered_map<E, std::string> reverseMap = []() {
561 std::unordered_map<E, std::string> result;
562 for (const auto& element : EnumStringMapping<E>::stringToEnum()) {
563 result.emplace(element.second, element.first);
564 }
565 return result;
566 }();
567 return reverseMap;
568 }
569};
