43struct StaffKey

47

64struct StaffKeyHash
65{
70 size_t operator()(const StaffKey& k) const noexcept
71 {
72 size_t h1 = std::hash<std::string>{}(k.partId);
73 size_t h2 = std::hash<int>{}(k.staffNo);
74 return h1 ^ (h2 + 0x9e3779b9u + (h1 << 6) + (h1 >> 2));
75 }
76};

93[[nodiscard]] inline std::optional<LayoutStaffKeySet>
94analyzeLayoutStaffVoices(const layout::Staff& staff)
95{
96 const auto sources = staff.sources();
97 if (sources.empty()) {
98 return std::nullopt;
99 }
100
101 struct KeyState
102 {
103 size_t count = 0;
104 bool anyMissingVoice = false;
105 std::unordered_set<std::string> voices;
106 };
107
108 std::unordered_map<StaffKey, KeyState, StaffKeyHash> stateByKey;
109 stateByKey.reserve(sources.size());
110
111 for (const auto src : sources) {
112 const std::string partId = src.part();
113 if (partId.empty()) {
114 return std::nullopt;
115 }
116
117 const StaffKey key{partId, src.staff()};
118 auto& st = stateByKey[key];
119 ++st.count;
120
121 const auto v = src.voice();
122 if (!v) {
123 st.anyMissingVoice = true;
124 continue;
125 }
126
127 if (v->empty()) {
128 return std::nullopt;
129 }
130
131 if (!st.voices.emplace(*v).second) {
132 return std::nullopt; // duplicate voice for same StaffKey
133 }
134 }
135
136 // Enforce per-StaffKey semantic rule.
137 for (const auto& kv : stateByKey) {
138 const KeyState& st = kv.second;
139 if (st.count > 1) {
140 if (st.anyMissingVoice || st.voices.size() != st.count) {
141 return std::nullopt;
142 }
143 }
144 }
145
146 LayoutStaffKeySet result;
147 result.reserve(stateByKey.size());
148 for (const auto& kv : stateByKey) {
149 result.insert(kv.first);
150 }
151
152 return result;
153}

164[[nodiscard]] inline std::optional<std::vector<layout::Staff>>
165flattenLayoutStaves(const Layout& layout)
166{
167 auto content = layout.content();
168 std::vector<layout::Staff> result;
169 result.reserve(content.size()); // lower bound; groups may expand further
170
171 const auto walk = [&](auto&& self, const ContentArray& content) -> std::optional<bool> {
172 for (auto elem : content) {
173 if (elem.type() == layout::Group::ContentTypeValue) {
174 layout::Group g = elem.get<layout::Group>();
175 auto ok = self(self, g.content());
176 if (!ok) {
177 return std::nullopt;
178 }
179 } else if (elem.type() == layout::Staff::ContentTypeValue) {
180 result.push_back(elem.get<layout::Staff>());
181 } else {
182 return std::nullopt;
183 }
184 }
185 return true;
186 };
187
188 if (!walk(walk, content)) {
189 return std::nullopt;
190 }
191 return result;
192}

298[[nodiscard]] inline std::optional<std::vector<LayoutSpan>>
299buildLayoutSpans(const mnx::Layout& layout)
300{
301 const auto content = layout.content();
302 std::vector<LayoutSpan> spans;
303 spans.reserve(content.size()); // lower bound
304
305 size_t staffIndex = 0;
306 size_t encounter = 0; // stable tiebreaker
307
308 struct SortKey { size_t start, depth, encounter; };
309
310 struct TaggedSpan {
311 LayoutSpan span;
312 SortKey key;
313 };
314
315 std::vector<TaggedSpan> tagged;
316 tagged.reserve(content.size());
317
318 // Return value:
319 // - outer std::optional: hard failure (unsupported element) if empty
320 // - inner std::optional: no staves in subtree if empty; otherwise first/last staff indices
321 const auto walk =
322 [&](auto&& self, const ContentArray& arr, size_t depth)
323 -> std::optional<std::optional<std::pair<size_t,size_t>>>
324 {
325 std::optional<size_t> first;
326 std::optional<size_t> last;
327
328 for (auto elem : arr) {
329 if (elem.type() == layout::Staff::ContentTypeValue) {
330 layout::Staff s = elem.get<layout::Staff>();
331
332 const size_t i = staffIndex++;
333
334 LayoutSpan span;
335 span.kind = LayoutSpan::Kind::Staff;
336 span.depth = depth + 1; // staff spans must be at the deepest depth
337 span.startIndex = i;
338 span.endIndex = i;
339 span.symbol = s.symbol();
340 span.label = s.label();
341 span.labelref = s.labelref();
342 span.sources = util::analyzeLayoutStaffVoices(s);
343
344 tagged.push_back({ std::move(span), SortKey{i, depth + 1, encounter++} });
345
346 first = first.value_or(i);
347 last = i;
348 } else if (elem.type() == layout::Group::ContentTypeValue) {
349 layout::Group g = elem.get<layout::Group>();
350
351 auto childRange = self(self, g.content(), depth + 1);
352 if (!childRange) {
353 return std::nullopt; // hard failure
354 }
355 if (!*childRange) {
356 continue; // skip groups with no staves anywhere in their subtree
357 }
358 const auto [cFirst, cLast] = **childRange;
359
360 LayoutSpan span;
361 span.kind = LayoutSpan::Kind::Group;
362 span.depth = depth;
363 span.startIndex = cFirst;
364 span.endIndex = cLast;
365 span.symbol = g.symbol();
366 span.label = g.label();
367
368 tagged.push_back({ std::move(span), SortKey{cFirst, depth, encounter++} });
369
370 first = first.value_or(cFirst);
371 last = cLast;
372 } else {
373 return std::nullopt; // unsupported content element
374 }
375 }
376
377 if (!first || !last) {
378 return std::optional<std::pair<size_t,size_t>>{}; // success, but no staves
379 }
380 return std::make_pair(*first, *last);
381 };
382
383 auto rootRange = walk(walk, content, /*depth*/0);
384 if (!rootRange) {
385 return std::nullopt;
386 }
387 // Empty root content is allowed; it simply produces an empty spans vector.
388
389 std::stable_sort(tagged.begin(), tagged.end(),
390 [](const TaggedSpan& a, const TaggedSpan& b)
391 {
392 if (a.key.start != b.key.start) return a.key.start < b.key.start;
393 if (a.key.depth != b.key.depth) return a.key.depth < b.key.depth;
394 return a.key.encounter < b.key.encounter;
395 });
396
397 spans.reserve(tagged.size());
398 for (auto& t : tagged) spans.push_back(std::move(t.span));
399 return spans;
400}

417[[nodiscard]] inline std::vector<LayoutSpan>
418buildDefaultLayoutSpans(const Array<Part>& parts)
419{
420 std::vector<LayoutSpan> result;
421 size_t staffIdx = 0;
422
423 for (const auto& part : parts) {
424 const size_t numStaves = static_cast<size_t>(part.staves());
425 if (numStaves == 0) {
426 continue;
427 }
428 size_t staffDepth = 1;
429 bool staffNameNeeded = true;
430 if (numStaves > 1) {
431 LayoutSpan groupSpan;
432 groupSpan.depth = staffDepth++;
433 groupSpan.kind = LayoutSpan::Kind::Group;
434 groupSpan.symbol = LayoutSymbol::Brace;
435 groupSpan.startIndex = staffIdx;
436 groupSpan.endIndex = staffIdx + numStaves - 1;
437 groupSpan.label = part.name();
438 staffNameNeeded = false;
439 result.emplace_back(std::move(groupSpan));
440 }
441 for (size_t x = 0; x < numStaves; x++) {
442 LayoutSpan staffSpan;
443 staffSpan.depth = staffDepth;
444 staffSpan.kind = LayoutSpan::Kind::Staff;
445 staffSpan.startIndex = staffIdx;
446 staffSpan.endIndex = staffIdx;
447 if (staffNameNeeded) {
448 staffSpan.label = part.name();
449 }
450 result.emplace_back(std::move(staffSpan));
451 staffIdx++;
452 }
453 }
454
455 return result;
456}
