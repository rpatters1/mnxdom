222class Base
223{
224public:
225 virtual ~Base() = default;
226
228 Base(const Base& src) : m_root(src.m_root), m_pointer(src.m_pointer)
229 {}
230
232 Base(Base&& src) noexcept : m_root(src.m_root), // m_root must be copied (not moved)
233 m_pointer(std::move(src.m_pointer))
234 {}
235
237 Base& operator=(const Base& src)
238 {
239 if (this != &src) {
240 if (m_root != src.m_root) {
241 throw std::logic_error("Assignment from a different JSON document is not allowed.");
242 }
243 m_pointer = src.m_pointer;
244 }
245 return *this;
246 }
247
249 Base& operator=(Base&& src)
250 {
251 if (this != &src) {
252 if (m_root != src.m_root) {
253 throw std::logic_error("Assignment from a different JSON document is not allowed.");
254 }
255 m_pointer = std::move(src.m_pointer);
256 }
257 return *this;
258 }
259
262 [[nodiscard]] std::string dump(int indents = -1) const
263 {
264 return ref().dump(indents);
265 }
266
270 template <typename T>
271 [[nodiscard]] T parent() const
272 {
273 static_assert(std::is_base_of_v<Base, T>, "Template type mush be derived from Base.");
274 return T(m_root, m_pointer.parent_pointer());
275 }
276
281 template <typename T, typename Scope = scope::Default>
282 [[nodiscard]] std::optional<T> getEnclosingElement() const;
283
285 [[nodiscard]] json_pointer pointer() const { return m_pointer; }
286
288 [[nodiscard]] Document document() const;
289
290protected:
296 [[nodiscard]] json& ref() const { return resolve_pointer(); }
297
302 [[nodiscard]] json& ref() { return resolve_pointer(); }
303
305 [[nodiscard]] const std::shared_ptr<json>& root() const { return m_root; }
306
312 Base(const std::shared_ptr<json>& root, json_pointer pointer)
313 : m_root(root), m_pointer(std::move(pointer)) {}
314
321 Base(json&& jsonRef, Base& parent, std::string_view key)
322 : m_root(parent.m_root), m_pointer(parent.m_pointer / std::string(key))
323 {
324 (*m_root)[m_pointer] = std::move(jsonRef);
325 }
326
334 template <typename T>
335 [[nodiscard]] T getChild(std::string_view key) const
336 {
337 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
338
339 json_pointer childPointer = m_pointer / std::string(key);
340 if (!checkKeyIsValid<T>(childPointer)) {
341 throw std::out_of_range("Child node not found: " + std::string(key));
342 }
343
344 return T(m_root, childPointer);
345 }
346
354 template <typename T>
355 T setChild(std::string_view key, const T& value)
356 {
357 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
358
359 json_pointer childPointer = m_pointer / std::string(key);
360 (*m_root)[childPointer] = value.ref();
361 return T(m_root, childPointer);
362 }
363
371 template <typename T>
372 [[nodiscard]] std::optional<T> getOptionalChild(std::string_view key) const
373 {
374 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
375
376 json_pointer childPointer = m_pointer / std::string(key);
377 if (!checkKeyIsValid<T>(childPointer)) {
378 return std::nullopt;
379 }
380
381 return T(m_root, childPointer);
382 }
383
384private:
392 template <typename T>
393 [[nodiscard]] bool checkKeyIsValid(const json_pointer& pointer) const
394 {
395 if (!(*m_root).contains(pointer)) {
396 return false;
397 }
398
399 const json& node = (*m_root).at(pointer);
400
401 if constexpr (std::is_base_of_v<Object, T>) {
402 if (!node.is_object()) {
403 throw std::runtime_error("Expected an object for: " + pointer.to_string());
404 }
405 } else if constexpr (std::is_base_of_v<Array<typename T::value_type>, T>) {
406 if (!node.is_array()) {
407 throw std::runtime_error("Expected an array for: " + pointer.to_string());
408 }
409 }
410
411 return true;
412 }
413
419 [[nodiscard]] json& resolve_pointer() const
420 {
421 return (*m_root).at(m_pointer); // Throws if invalid
422 }
423
424 const std::shared_ptr<json> m_root;
425 json_pointer m_pointer;
426
427 friend class validation::SemanticValidator;
428};

516 public:
517 using iterator_category = std::input_iterator_tag;

801template <typename E, typename = std::enable_if_t<std::is_enum_v<E>>>
802struct EnumStringMapping
803{
804 static const std::unordered_map<std::string, E> stringToEnum();
805
807 static const std::unordered_map<E, std::string> enumToString()
808 {
809 static const std::unordered_map<E, std::string> reverseMap = []() {
810 std::unordered_map<E, std::string> result;
811 for (const auto& element : EnumStringMapping<E>::stringToEnum()) {
812 result.emplace(element.second, element.first);
813 }
814 return result;
815 }();
816 return reverseMap;
817 }

823template <typename T>
824class Dictionary : public Object
825{
826 static_assert(std::is_arithmetic_v<T> || std::is_same_v<T, std::string> ||
827 std::is_base_of_v<ArrayElementObject, T>, "Invalid MNX dictionary element type.");
828
829private:
830 template <typename DictionaryType, typename IteratorType>
831 struct iter
832 {
833 using value_type = std::pair<const std::string, T>;
834 using difference_type = std::ptrdiff_t;
835 using iterator_category = std::forward_iterator_tag;
836 using pointer = value_type*;
837 using reference = value_type&;
838
839 private:
840 DictionaryType* m_ptr;
841 IteratorType m_it;
842 mutable std::unique_ptr<value_type> m_pair; // Cached key-value pair for operator*() and operator->()
843
844 void update_pair() const {
845 m_pair.reset();
846 if (m_it != m_ptr->ref().end()) {
847 m_pair = std::make_unique<value_type>(m_it.key(), m_ptr->valueForKey(m_it.key()));
848 }
849 }
850
851 public:
852 iter(DictionaryType* ptr, IteratorType it) : m_ptr(ptr), m_it(it)
853 { update_pair(); }
854
855 [[nodiscard]] reference operator*() const { return *m_pair.get(); }
856 [[nodiscard]] pointer operator->() const { return m_pair.get(); }
857
858 iter& operator++() { ++m_it; update_pair(); return *this; }
859 iter operator++(int) { iter tmp = *this; ++(*this); return tmp; }
860
861 [[nodiscard]] bool operator!=(const iter& o) const { return m_it != o.m_it; }
862 [[nodiscard]] bool operator==(const iter& o) const { return m_it == o.m_it; }
863 };

867 using value_type = T;

869 using iterator = iter<Dictionary, json::iterator>;

987template<typename EnumType>

992 {

995 auto it = map.find(j.get<std::string>());
996 if (it != map.end()) {
997 return it->second;
998 }
1000 return EnumType{};
1001 }

1004 static void to_json(BasicJsonType& j, const EnumType& value)
1005 {
1006 const auto& map = ::mnx::EnumStringMapping<EnumType>::enumToString();
1007 auto it = map.find(value);
1008 if (it == map.end()) {
1010 j = BasicJsonType();
1011 return;
1012 }
1013 j = it->second;
1014 }
