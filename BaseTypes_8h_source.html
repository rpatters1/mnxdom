120 template <typename T>
121 T parent() const
122 {
123 static_assert(std::is_base_of_v<Base, T>, "Template type mush be derived from Base.");
124 return T(m_root, m_pointer.parent_pointer());

132 Base(const std::reference_wrapper<json>& root, json_pointer pointer)
133 : m_root(root), m_pointer(std::move(pointer)) {}
134
141 Base(json&& jsonRef, Base& parent, const std::string_view& key)
142 : m_root(parent.m_root), m_pointer(parent.m_pointer / std::string(key))
143 {
144 m_root.get()[m_pointer] = std::move(jsonRef);
145 }
146
154 template <typename T>
155 T getChild(const std::string_view& key) const
156 {
157 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
158
159 json_pointer childPointer = m_pointer / std::string(key);
160 if (!checkKeyIsValid<T>(childPointer)) {
161 throw std::runtime_error("Missing required child node: " + std::string(key));
162 }
163
164 return T(m_root, childPointer);
165 }
166
174 template <typename T>
175 T setChild(const std::string_view& key, const T& value)
176 {
177 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
178
179 json_pointer childPointer = m_pointer / std::string(key);
180 m_root.get()[childPointer] = value.ref();
181 return T(m_root, childPointer);
182 }
183
191 template <typename T>
192 std::optional<T> getOptionalChild(const std::string_view& key) const
193 {
194 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
195
196 json_pointer childPointer = m_pointer / std::string(key);
197 if (!checkKeyIsValid<T>(childPointer)) {
198 return std::nullopt;
199 }
200
201 return T(m_root, childPointer);
202 }
203
204private:
212 template <typename T>
213 bool checkKeyIsValid(const json_pointer& pointer) const
214 {
215 if (!m_root.get().contains(pointer)) {
216 return false;
217 }
218
219 const json& node = m_root.get().at(pointer);
220
221 if constexpr (std::is_base_of_v<Object, T>) {
222 if (!node.is_object()) {
223 throw std::runtime_error("Expected an object for: " + pointer.to_string());
224 }
225 } else if constexpr (std::is_base_of_v<Array<typename T::value_type>, T>) {
226 if (!node.is_array()) {
227 throw std::runtime_error("Expected an array for: " + pointer.to_string());
228 }
229 }
230
231 return true;
232 }
233
239 json& resolve_pointer() const
240 {
241 return m_root.get().at(m_pointer); // Throws if invalid
242 }
243
244 std::reference_wrapper<json> m_root;
245 json_pointer m_pointer;
246};

469 {
470 this->checkIndex(index);
471 auto element = (*this)[index];
472 if (element.type() != T::ContentTypeValue) {
473 throw std::invalid_argument("Type mismatch: expected " + std::string(T::ContentTypeValue) +

484template <typename E, typename = std::enable_if_t<std::is_enum_v<E>>>
485struct EnumStringMapping
486{
487 static const std::unordered_map<std::string, E> stringToEnum();
488
490 static const std::unordered_map<E, std::string> enumToString()
491 {
492 static const std::unordered_map<E, std::string> reverseMap = []() {
493 std::unordered_map<E, std::string> result;
494 for (const auto& element : EnumStringMapping<E>::stringToEnum()) {
495 result.emplace(element.second, element.first);
496 }
497 return result;
498 }();
499 return reverseMap;
500 }
501};

512template<typename EnumType>

514{

518 // Lookup the string in the specialized map.

520 auto it = map.find(j.get<std::string>());
521 if (it != map.end()) {

523 }
525 return EnumType{};
526 }

533 if (it == map.end()) {
535 j = BasicJsonType();

538 j = it->second;
539 }
