30class Fraction : private Array<unsigned int>
31{
32private:

36 static constexpr size_t NUMERATOR_INDEX = 0;
37 static constexpr size_t DENOMINATOR_INDEX = 1;
38
39public:
41 Fraction(const std::shared_ptr<json>& root, json_pointer pointer)
42 : ArrayType(root, pointer)
43 {
44 if (size() != 2) {
45 throw std::invalid_argument("mnx::Fraction must have exactly 2 elements.");
46 }
47 }
48
54 Fraction(Base& parent, const std::string_view& key, NumType numerator, NumType denominator)
55 : ArrayType(parent, key)
56 {
57 push_back(numerator);
58 push_back(denominator);
59 }
60
61 MNX_ARRAY_ELEMENT_PROPERTY(NumType, numerator, NUMERATOR_INDEX);
62 MNX_ARRAY_ELEMENT_PROPERTY(NumType, denominator, DENOMINATOR_INDEX);
