180class Base
181{
182public:
183 virtual ~Base() = default;
184
186 Base(const Base& src) : m_root(src.m_root), m_pointer(src.m_pointer)
187 {}
188
190 Base(Base&& src) noexcept : m_root(src.m_root), // m_root must be copied (not moved)
191 m_pointer(std::move(src.m_pointer))
192 {}
193
195 Base& operator=(const Base& src)
196 {
197 if (this != &src) {
198 if (m_root != src.m_root) {
199 throw std::logic_error("Assignment from a different JSON document is not allowed.");
200 }
201 m_pointer = src.m_pointer;
202 }
203 return *this;
204 }
205
207 Base& operator=(Base&& src)
208 {
209 if (this != &src) {
210 if (m_root != src.m_root) {
211 throw std::logic_error("Assignment from a different JSON document is not allowed.");
212 }
213 m_pointer = std::move(src.m_pointer);
214 }
215 return *this;
216 }
217
220 [[nodiscard]] std::string dump(int indents = -1) const
221 {
222 return ref().dump(indents);
223 }
224
228 template <typename T>
229 [[nodiscard]] T parent() const
230 {
231 static_assert(std::is_base_of_v<Base, T>, "Template type mush be derived from Base.");
232 return T(m_root, m_pointer.parent_pointer());
233 }
234
239 template <typename T, typename Scope = scope::Default>
240 [[nodiscard]] std::optional<T> getEnclosingElement() const;
241
243 [[nodiscard]] json_pointer pointer() const { return m_pointer; }
244
246 [[nodiscard]] Document document() const;
247
248protected:
254 [[nodiscard]] json& ref() const { return resolve_pointer(); }
255
260 [[nodiscard]] json& ref() { return resolve_pointer(); }
261
263 [[nodiscard]] const std::shared_ptr<json>& root() const { return m_root; }
264
270 Base(const std::shared_ptr<json>& root, json_pointer pointer)
271 : m_root(root), m_pointer(std::move(pointer)) {}
272
279 Base(json&& jsonRef, Base& parent, std::string_view key)
280 : m_root(parent.m_root), m_pointer(parent.m_pointer / std::string(key))
281 {
282 (*m_root)[m_pointer] = std::move(jsonRef);
283 }
284
292 template <typename T>
293 [[nodiscard]] T getChild(std::string_view key) const
294 {
295 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
296
297 json_pointer childPointer = m_pointer / std::string(key);
298 if (!checkKeyIsValid<T>(childPointer)) {
299 throw std::out_of_range("Child node not found: " + std::string(key));
300 }
301
302 return T(m_root, childPointer);
303 }
304
312 template <typename T>
313 T setChild(std::string_view key, const T& value)
314 {
315 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
316
317 json_pointer childPointer = m_pointer / std::string(key);
318 (*m_root)[childPointer] = value.ref();
319 return T(m_root, childPointer);
320 }
321
329 template <typename T>
330 [[nodiscard]] std::optional<T> getOptionalChild(std::string_view key) const
331 {
332 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
333
334 json_pointer childPointer = m_pointer / std::string(key);
335 if (!checkKeyIsValid<T>(childPointer)) {
336 return std::nullopt;
337 }
338
339 return T(m_root, childPointer);
340 }
341
342private:
350 template <typename T>
351 [[nodiscard]] bool checkKeyIsValid(const json_pointer& pointer) const
352 {
353 if (!(*m_root).contains(pointer)) {
354 return false;
355 }
356
357 const json& node = (*m_root).at(pointer);
358
359 if constexpr (std::is_base_of_v<Object, T>) {
360 if (!node.is_object()) {
361 throw std::runtime_error("Expected an object for: " + pointer.to_string());
362 }
363 } else if constexpr (std::is_base_of_v<Array<typename T::value_type>, T>) {
364 if (!node.is_array()) {
365 throw std::runtime_error("Expected an array for: " + pointer.to_string());
366 }
367 }
368
369 return true;
370 }
371
377 [[nodiscard]] json& resolve_pointer() const
378 {
379 return (*m_root).at(m_pointer); // Throws if invalid
380 }
381
382 const std::shared_ptr<json> m_root;
383 json_pointer m_pointer;
384
385 friend class validation::SemanticValidator;
386};

463{

467private:
468 template<typename ArrayType>
469 struct iter
470 {
471 private:
472 ArrayType* m_ptr;
473 mutable size_t m_idx;
474
475 public:
476 using iterator_category = std::input_iterator_tag;
477 using value_type = T;
478 using difference_type = std::ptrdiff_t;
479 using pointer = void; // not meaningful for proxy/value-returning iterators
480 using reference = T; // since operator* returns by value
481
482 iter(ArrayType* ptr, size_t idx) : m_ptr(ptr), m_idx(idx) {}
483 T operator*() const { return (*m_ptr)[m_idx]; }
484 iter& operator++() { ++m_idx; return *this; }
485 bool operator!=(const iter& o) const { return m_idx != o.m_idx; }
486 };

488public:
490 using value_type = T;

492 using iterator = iter<Array>;
493 using const_iterator = iter<const Array>;

498 Array(const std::shared_ptr<json>& root, json_pointer pointer) : Base(root, pointer)
499 {
500 if (!ref().is_array()) {
501 throw std::invalid_argument("mnx::Array must wrap a JSON array.");
502 }
503 }
504
508 Array(Base& parent, std::string_view key)
509 : Base(json::array(), parent, key) {}
510
512 [[nodiscard]] size_t size() const { return ref().size(); }
513
515 [[nodiscard]] bool empty() const { return ref().empty(); }
516
518 void clear() { ref().clear(); }
519
521 [[nodiscard]] T at(size_t index) const
522 {
523 return operator[](index);
524 }
525
529 [[nodiscard]] T front() const { return (*this)[0]; }
530
534 [[nodiscard]] T back() const { return (*this)[size() - 1]; }
535
537 [[nodiscard]] auto operator[](size_t index) const
538 {
539 checkIndex(index);
540 if constexpr (std::is_base_of_v<Base, T>) {
541 return getChild<T>(std::to_string(index));
542 } else {
543 return getChild<SimpleType<T>>(std::to_string(index));
544 }
545 }
546
548 [[nodiscard]] auto operator[](size_t index)
549 {
550 checkIndex(index);
551 if constexpr (std::is_base_of_v<Base, T>) {
552 return getChild<T>(std::to_string(index));
553 } else {
554 return getChild<SimpleType<T>>(std::to_string(index));
555 }
556 }

580 { return std::vector<U>(begin(), end()); }
581
583 [[nodiscard]] auto begin() { return iterator(this, 0); }
584
586 [[nodiscard]] auto end() { return iterator(this, size()); }
587
589 [[nodiscard]] auto begin() const { return const_iterator(this, 0); }
590
592 [[nodiscard]] auto end() const { return const_iterator(this, size()); }
593
594protected:
597 void checkIndex(size_t index) const
598 {
599 assert(index < ref().size());
600 if (index >= ref().size()) {
601 throw std::out_of_range("Index out of range");
602 }
603 }
604
605private:
606 template <typename U, typename... Args>
607 U appendImpl(Args&&... args)
608 {
609 static_assert(std::is_base_of_v<Base, U>, "Array::appendImpl requires a Base-derived element type.");
610 if constexpr (std::is_base_of_v<Object, U>) {
611 ref().push_back(json::object());
612 } else {
613 ref().push_back(json::array());
614 }
615 return U(*this, std::to_string(ref().size() - 1), std::forward<Args>(args)...);
616 }

794template <typename E, typename = std::enable_if_t<std::is_enum_v<E>>>
795struct EnumStringMapping
796{
797 static const std::unordered_map<std::string, E> stringToEnum();
798
800 static const std::unordered_map<E, std::string> enumToString()
801 {
802 static const std::unordered_map<E, std::string> reverseMap = []() {
803 std::unordered_map<E, std::string> result;
804 for (const auto& element : EnumStringMapping<E>::stringToEnum()) {
805 result.emplace(element.second, element.first);
806 }
807 return result;
808 }();
809 return reverseMap;
810 }

816template <typename T>
817class Dictionary : public Object,
818 public detail::DictionaryAppendOverloads<Dictionary<T>, T>
819{
820 static_assert(std::is_arithmetic_v<T> || std::is_same_v<T, std::string> ||
821 std::is_base_of_v<ArrayElementObject, T>, "Invalid MNX dictionary element type.");
822
823private:
824 template <typename DictionaryType, typename IteratorType>
825 struct iter
826 {
827 using value_type = std::pair<const std::string, T>;
828 using difference_type = std::ptrdiff_t;
829 using iterator_category = std::forward_iterator_tag;
830 using pointer = value_type*;
831 using reference = value_type&;
832
833 private:
834 DictionaryType* m_ptr {};
835 IteratorType m_it {};
836 mutable std::optional<value_type> m_pair; // cached key/value
837
838 void update_pair() const
839 {
840 m_pair.reset();
841 if (m_it != m_ptr->ref().end()) {
842 m_pair.emplace(m_it.key(), m_ptr->valueForKey(m_it.key()));
843 }
844 }
845
846 public:
847 iter(DictionaryType* ptr, IteratorType it)
848 : m_ptr(ptr), m_it(it)
849 {
850 update_pair();
851 }
852
853 [[nodiscard]] reference operator*() const { return *m_pair; }
854 [[nodiscard]] pointer operator->() const { return &*m_pair; }
855
856 iter& operator++() { ++m_it; update_pair(); return *this; }
857 iter operator++(int) { iter tmp(*this); ++(*this); return tmp; }
858
859 [[nodiscard]] bool operator!=(const iter& o) const { return m_it != o.m_it; }
860 [[nodiscard]] bool operator==(const iter& o) const { return m_it == o.m_it; }
861 };
862
863public:
865 using value_type = T;
866
867 using iterator = iter<Dictionary, json::iterator>;
868 using const_iterator = iter<const Dictionary, json::const_iterator>;
869
