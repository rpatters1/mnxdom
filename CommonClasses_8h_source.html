72struct [[nodiscard]] FractionValue
73{
75 using NumType = unsigned;
76 static_assert(std::is_integral<NumType>::value, "FractionValue::NumType must be an integral type");
77
78private:
79 NumType m_num = 0;
80 NumType m_den = 1;
81
82 template<typename T = NumType,
83 typename std::enable_if_t<std::is_signed<T>::value, int> = 0>
84 constexpr void normalizeSign()
85 {
86 if (m_den < 0) {
87 m_den = -m_den;
88 m_num = -m_num;
89 }
90 }
91
92 // Unsigned case: no-op, never even sees m_den < 0
93 template<typename T = NumType,
94 typename std::enable_if_t<!std::is_signed<T>::value, int> = 0>
95 constexpr void normalizeSign() {}
96
97public:
101 constexpr FractionValue() = default;
102
115 constexpr FractionValue(NumType num, NumType den)
116 : m_num(num), m_den(den)
117 {
118 if (m_den == 0) {
119 throw std::invalid_argument("FractionValue: denominator must not be zero.");
120 }
121 normalizeSign();
122 }
123
129 constexpr FractionValue(NumType value) : m_num(value), m_den(1) {}
130
132 [[nodiscard]] constexpr NumType numerator() const noexcept { return m_num; }
133
135 [[nodiscard]] constexpr NumType denominator() const noexcept { return m_den; }
136
141 [[nodiscard]] constexpr NumType quotient() const
142 {
143 return m_num / m_den;
144 }
145
150 constexpr FractionValue remainder() const
151 {
152 FractionValue result;
153 result.m_num = m_num % m_den;
154 result.m_den = m_den;
155 return result;
156 }
157
159 static constexpr FractionValue max() noexcept
160 {
161 return FractionValue((std::numeric_limits<NumType>::max)());
162 }
163
165 [[nodiscard]] constexpr double toDouble() const {
166 return double(m_num) / double(m_den);
167 }
168
177 constexpr FractionValue& operator+=(const FractionValue& rhs)
178 {
179 // a/b + c/d = (ad + bc)/bd
180 m_num = m_num * rhs.m_den + rhs.m_num * m_den;
181 m_den = m_den * rhs.m_den;
182 reduce();
183 return *this;
184 }
185
194 constexpr FractionValue& operator-=(const FractionValue& rhs)
195 {
196 // a/b - c/d = (ad - bc)/bd
197 m_num = m_num * rhs.m_den - rhs.m_num * m_den;
198 m_den = m_den * rhs.m_den;
199 reduce();
200 return *this;
201 }
202
211 constexpr FractionValue& operator*=(const FractionValue& rhs)
212 {
213 m_num = m_num * rhs.m_num;
214 m_den = m_den * rhs.m_den;
215 reduce();
216 return *this;
217 }
218
229 constexpr FractionValue& operator/=(const FractionValue& rhs)
230 {
231 if (rhs.m_num == 0) {
232 throw std::invalid_argument("Division by zero FractionValue.");
233 }
234 m_num = m_num * rhs.m_den;
235 m_den = m_den * rhs.m_num;
236 reduce();
237 return *this;
238 }
239
243 constexpr void reduce()
244 {
245 const NumType g = std::gcd(m_num, m_den);
246 if (g > 1) {
247 m_num /= g;
248 m_den /= g;
249 }
250 normalizeSign();
251 }
252
256 constexpr FractionValue reduced() const
257 {
258 auto result = *this;
259 result.reduce();
260 return result;
261 }
262
293 constexpr bool expressWithDenominator(NumType targetDenominator)
294 {
295 if (targetDenominator == 0) {
296 // Cannot express anything with denominator 0.
297 return false;
298 }
299
300 // Zero fraction: 0/d is expressible as 0/targetDenominator.
301 if (m_num == 0) {
302 m_den = targetDenominator;
303 return true;
304 }
305
306 // Reduce the fraction logically: m_num/m_den -> (numRed/denRed).
307 // We don't actually have to store the reduced form; just use it to
308 // check compatibility and compute the new numerator.
309 const NumType g = std::gcd(m_num, m_den);
310 const NumType denRed = m_den / g;
311 const NumType numRed = m_num / g; // sign preserved
312
313 // For an integer representation with the requested denominator to exist
314 // (while preserving the value), the reduced denominator must divide
315 // targetDenominator exactly.
316 if (targetDenominator % denRed != 0) {
317 return false; // No integer scaling factor exists.
318 }
319
320 const NumType factor = targetDenominator / denRed;
321
322 // Apply the scaling to the reduced numerator and store the new form.
323 m_num = numRed * factor;
324 m_den = targetDenominator;
325
326 normalizeSign();
327
328 return true;
329 }

404
405 static constexpr size_t NUMERATOR_INDEX = 0;
406 static constexpr size_t DENOMINATOR_INDEX = 1;
407
408public:
410 Fraction(const std::shared_ptr<json>& root, json_pointer pointer)
411 : ArrayType(root, pointer)
412 {
413 if (size() != 2) {
414 throw std::invalid_argument("mnx::Fraction must have exactly 2 elements.");
415 }
416 }
417
422 Fraction(Base& parent, std::string_view key, const FractionValue& value)

425 push_back(value.numerator());

428

432 return FractionValue(numerator(), denominator());
433 }
434

436 MNX_ARRAY_ELEMENT_PROPERTY(NumType, denominator, DENOMINATOR_INDEX);
