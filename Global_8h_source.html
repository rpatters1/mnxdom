40 size_t calcDefaultMeasureIndex() const
41 {
42 size_t arrayIndex = calcArrayIndex();
43 if (arrayIndex == 0) return 1;
44 auto parents = parent<Array<GlobalMeasure>>();
45 auto prev = parents[arrayIndex - 1];
46 auto prevIndex = prev.index();
47 return prevIndex.has_value() ? *prevIndex : prev.calcDefaultMeasureIndex();
48 }
