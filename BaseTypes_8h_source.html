210class Base
211{
212public:
213 virtual ~Base() = default;
214
216 Base(const Base& src) : m_root(src.m_root), m_pointer(src.m_pointer)
217 {}
218
220 Base(Base&& src) noexcept : m_root(src.m_root), // m_root must be copied (not moved)
221 m_pointer(std::move(src.m_pointer))
222 {}
223
225 Base& operator=(const Base& src)
226 {
227 if (this != &src) {
228 if (m_root != src.m_root) {
229 throw std::logic_error("Assignment from a different JSON document is not allowed.");
230 }
231 m_pointer = src.m_pointer;
232 }
233 return *this;
234 }
235
237 Base& operator=(Base&& src)
238 {
239 if (this != &src) {
240 if (m_root != src.m_root) {
241 throw std::logic_error("Assignment from a different JSON document is not allowed.");
242 }
243 m_pointer = std::move(src.m_pointer);
244 }
245 return *this;
246 }
247
250 std::string dump(int indents = -1)
251 {
252 return ref().dump(indents);
253 }
254
258 template <typename T>
259 T parent() const
260 {
261 static_assert(std::is_base_of_v<Base, T>, "Template type mush be derived from Base.");
262 return T(m_root, m_pointer.parent_pointer());
263 }
264
266 std::optional<Part> getPart();
267
268protected:
274 json& ref() const { return resolve_pointer(); }
275
280 json& ref() { return resolve_pointer(); }
281
283 const std::shared_ptr<json>& root() const { return m_root; }
284
286 json_pointer pointer() const { return m_pointer; }
287
293 Base(const std::shared_ptr<json>& root, json_pointer pointer)
294 : m_root(root), m_pointer(std::move(pointer)) {}
295
302 Base(json&& jsonRef, Base& parent, const std::string_view& key)
303 : m_root(parent.m_root), m_pointer(parent.m_pointer / std::string(key))
304 {
305 (*m_root)[m_pointer] = std::move(jsonRef);
306 }
307
315 template <typename T>
316 T getChild(const std::string_view& key) const
317 {
318 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
319
320 json_pointer childPointer = m_pointer / std::string(key);
321 if (!checkKeyIsValid<T>(childPointer)) {
322 throw std::runtime_error("Missing required child node: " + std::string(key));
323 }
324
325 return T(m_root, childPointer);
326 }
327
335 template <typename T>
336 T setChild(const std::string_view& key, const T& value)
337 {
338 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
339
340 json_pointer childPointer = m_pointer / std::string(key);
341 (*m_root)[childPointer] = value.ref();
342 return T(m_root, childPointer);
343 }
344
352 template <typename T>
353 std::optional<T> getOptionalChild(const std::string_view& key) const
354 {
355 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
356
357 json_pointer childPointer = m_pointer / std::string(key);
358 if (!checkKeyIsValid<T>(childPointer)) {
359 return std::nullopt;
360 }
361
362 return T(m_root, childPointer);
363 }
364
365private:
373 template <typename T>
374 bool checkKeyIsValid(const json_pointer& pointer) const
375 {
376 if (!(*m_root).contains(pointer)) {
377 return false;
378 }
379
380 const json& node = (*m_root).at(pointer);
381
382 if constexpr (std::is_base_of_v<Object, T>) {
383 if (!node.is_object()) {
384 throw std::runtime_error("Expected an object for: " + pointer.to_string());
385 }
386 } else if constexpr (std::is_base_of_v<Array<typename T::value_type>, T>) {
387 if (!node.is_array()) {
388 throw std::runtime_error("Expected an array for: " + pointer.to_string());
389 }
390 }
391
392 return true;
393 }
394
400 json& resolve_pointer() const
401 {
402 return (*m_root).at(m_pointer); // Throws if invalid
403 }
404
405 const std::shared_ptr<json> m_root;
406 json_pointer m_pointer;
407
408 friend class validation::SemanticValidator;
409};

732template <typename E, typename = std::enable_if_t<std::is_enum_v<E>>>
733struct EnumStringMapping
734{
735 static const std::unordered_map<std::string, E> stringToEnum();
736
738 static const std::unordered_map<E, std::string> enumToString()
739 {
740 static const std::unordered_map<E, std::string> reverseMap = []() {
741 std::unordered_map<E, std::string> result;
742 for (const auto& element : EnumStringMapping<E>::stringToEnum()) {
743 result.emplace(element.second, element.first);
744 }
745 return result;
746 }();
747 return reverseMap;
748 }

754template <typename T>
755class Dictionary : public Object
756{
757 static_assert(std::is_arithmetic_v<T> || std::is_same_v<T, std::string> ||
758 std::is_base_of_v<ArrayElementObject, T>, "Invalid MNX dictionary element type.");
759
760private:
761 template <typename DictionaryType, typename IteratorType>
762 struct iter
763 {
764 using value_type = std::pair<const std::string, T>;
765 using difference_type = std::ptrdiff_t;
766 using iterator_category = std::forward_iterator_tag;
767 using pointer = value_type*;
768 using reference = value_type&;
769
770 private:
771 DictionaryType* m_ptr;
772 IteratorType m_it;
773 mutable std::unique_ptr<value_type> m_pair; // Cached key-value pair for operator*() and operator->()
774
775 void update_pair() const {
776 m_pair.reset();
777 if (m_it != m_ptr->ref().end()) {
778 m_pair = std::make_unique<value_type>(m_it.key(), m_ptr->operator[](m_it.key()));
779 }
780 }
781
782 public:
783 iter(DictionaryType* ptr, IteratorType it) : m_ptr(ptr), m_it(it)
784 { update_pair(); }
785
786 reference operator*() const { return *m_pair.get(); }
787 pointer operator->() const { return m_pair.get(); }
788
789 iter& operator++() { ++m_it; update_pair(); return *this; }
790 iter operator++(int) { iter tmp = *this; ++(*this); return tmp; }
791
792 bool operator!=(const iter& o) const { return m_it != o.m_it; }
793 bool operator==(const iter& o) const { return m_it == o.m_it; }
794 };
795
796public:

800 using iterator = iter<Dictionary, json::iterator>;
