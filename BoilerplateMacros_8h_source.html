111#define MNX_REQUIRED_PROPERTY(TYPE, NAME) \
112 [[nodiscard]] TYPE NAME() const { \
113 if (!ref().contains(#NAME)) { \
114 throw std::runtime_error("Missing required property: " #NAME); \
115 } \
116 return ref()[#NAME].get<TYPE>(); \
117 } \
118 void set_##NAME(const TYPE& value) { ref()[#NAME] = value; } \
119 static_assert(true, "") // require semicolon after macro

155 #define MNX_OPTIONAL_NAMED_PROPERTY(TYPE, NAME, KEY) \
156 [[nodiscard]] std::optional<TYPE> NAME() const { \
157 return ref().contains(KEY) ? std::optional<TYPE>(ref()[KEY].get<TYPE>()) : std::nullopt; \
158 } \
159 [[nodiscard]] TYPE NAME##_or(const TYPE& defaultVal) const { \
160 return ref().contains(KEY) ? ref()[KEY].get<TYPE>() : defaultVal; \
161 } \
162 void set_##NAME(const TYPE& value) { ref()[KEY] = value; } \
163 void clear_##NAME() { ref().erase(KEY); } \
164 static_assert(true, "") // require semicolon after macro

196#define MNX_OPTIONAL_PROPERTY_WITH_DEFAULT(TYPE, NAME, DEFAULT) \
197 [[nodiscard]] TYPE NAME() const { \
198 return ref().contains(#NAME) ? ref()[#NAME].get<TYPE>() : DEFAULT; \
199 } \
200 void set_##NAME(const TYPE& value) { ref()[#NAME] = value; } \
201 void clear_##NAME() { ref().erase(#NAME); } \
