200class Base
201{
202public:
203 virtual ~Base() = default;
204
206 Base(const Base& src) : m_root(src.m_root), m_pointer(src.m_pointer)
207 {}
208
210 Base(Base&& src) noexcept : m_root(src.m_root), // m_root must be copied (not moved)
211 m_pointer(std::move(src.m_pointer))
212 {}
213
215 Base& operator=(const Base& src)
216 {
217 if (this != &src) {
218 if (m_root != src.m_root) {
219 throw std::logic_error("Assignment from a different JSON document is not allowed.");
220 }
221 m_pointer = src.m_pointer;
222 }
223 return *this;
224 }
225
227 Base& operator=(Base&& src)
228 {
229 if (this != &src) {
230 if (m_root != src.m_root) {
231 throw std::logic_error("Assignment from a different JSON document is not allowed.");
232 }
233 m_pointer = std::move(src.m_pointer);
234 }
235 return *this;
236 }

240 std::string dump(int indents = -1)
241 {
242 return ref().dump(indents);
243 }

248 template <typename T>
249 T parent() const
250 {
251 static_assert(std::is_base_of_v<Base, T>, "Template type mush be derived from Base.");
252 return T(m_root, m_pointer.parent_pointer());
253 }
254
255protected:
261 json& ref() const { return resolve_pointer(); }
262
267 json& ref() { return resolve_pointer(); }
268
270 const std::shared_ptr<json>& root() const { return m_root; }
271
273 json_pointer pointer() const { return m_pointer; }
274
280 Base(const std::shared_ptr<json>& root, json_pointer pointer)
281 : m_root(root), m_pointer(std::move(pointer)) {}
282
289 Base(json&& jsonRef, Base& parent, const std::string_view& key)
290 : m_root(parent.m_root), m_pointer(parent.m_pointer / std::string(key))
291 {
292 (*m_root)[m_pointer] = std::move(jsonRef);
293 }

302 template <typename T>
303 T getChild(const std::string_view& key) const
304 {
305 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
306
307 json_pointer childPointer = m_pointer / std::string(key);
308 if (!checkKeyIsValid<T>(childPointer)) {
309 throw std::runtime_error("Missing required child node: " + std::string(key));
310 }
311
312 return T(m_root, childPointer);
313 }
314
322 template <typename T>
323 T setChild(const std::string_view& key, const T& value)
324 {
325 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
326
327 json_pointer childPointer = m_pointer / std::string(key);
328 (*m_root)[childPointer] = value.ref();
329 return T(m_root, childPointer);
330 }
331
339 template <typename T>
340 std::optional<T> getOptionalChild(const std::string_view& key) const
341 {
342 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
343
344 json_pointer childPointer = m_pointer / std::string(key);
345 if (!checkKeyIsValid<T>(childPointer)) {
346 return std::nullopt;
347 }
348
349 return T(m_root, childPointer);
350 }
351
352private:
360 template <typename T>
361 bool checkKeyIsValid(const json_pointer& pointer) const
362 {
363 if (!(*m_root).contains(pointer)) {
364 return false;
365 }
366
367 const json& node = (*m_root).at(pointer);
368
369 if constexpr (std::is_base_of_v<Object, T>) {
370 if (!node.is_object()) {
371 throw std::runtime_error("Expected an object for: " + pointer.to_string());
372 }
373 } else if constexpr (std::is_base_of_v<Array<typename T::value_type>, T>) {
374 if (!node.is_array()) {
375 throw std::runtime_error("Expected an array for: " + pointer.to_string());
376 }
377 }
378
379 return true;
380 }
381
387 json& resolve_pointer() const
388 {
389 return (*m_root).at(m_pointer); // Throws if invalid
390 }
391
392 const std::shared_ptr<json> m_root;
393 json_pointer m_pointer;
394};

705template <typename E, typename = std::enable_if_t<std::is_enum_v<E>>>
706struct EnumStringMapping
707{
708 static const std::unordered_map<std::string, E> stringToEnum();
709
711 static const std::unordered_map<E, std::string> enumToString()
712 {
713 static const std::unordered_map<E, std::string> reverseMap = []() {
714 std::unordered_map<E, std::string> result;
715 for (const auto& element : EnumStringMapping<E>::stringToEnum()) {
716 result.emplace(element.second, element.first);
717 }
718 return result;
719 }();
720 return reverseMap;
721 }

727template <typename T>
728class Dictionary : public Object
729{
730 static_assert(std::is_arithmetic_v<T> || std::is_same_v<T, std::string> ||
731 std::is_base_of_v<ArrayElementObject, T>, "Invalid MNX dictionary element type.");
732
733private:
734 template <typename DictionaryType, typename IteratorType>
735 struct iter
736 {
737 using value_type = std::pair<const std::string, T>;
738 using difference_type = std::ptrdiff_t;
739 using iterator_category = std::forward_iterator_tag;
740 using pointer = value_type*;
741 using reference = value_type&;
742
743 private:
744 DictionaryType* m_ptr;
745 IteratorType m_it;
746 mutable std::unique_ptr<value_type> m_pair; // Cached key-value pair for operator*() and operator->()
747
748 void update_pair() const {
749 m_pair.reset();
750 if (m_it != m_ptr->ref().end()) {
751 m_pair = std::make_unique<value_type>(m_it.key(), m_ptr->operator[](m_it.key()));
752 }
753 }
754
755 public:
756 iter(DictionaryType* ptr, IteratorType it) : m_ptr(ptr), m_it(it)
757 { update_pair(); }
758
759 reference operator*() const { return *m_pair.get(); }
760 pointer operator->() const { return m_pair.get(); }
761
762 iter& operator++() { ++m_it; update_pair(); return *this; }
763 iter operator++(int) { iter tmp = *this; ++(*this); return tmp; }
764
765 bool operator!=(const iter& o) const { return m_it != o.m_it; }
766 bool operator==(const iter& o) const { return m_it == o.m_it; }
767 };
768
769public:
771 using value_type = T;
772
773 using iterator = iter<Dictionary, json::iterator>;
774 using const_iterator = iter<const Dictionary, json::const_iterator>;
