72template <typename T, auto MakeFunc>
73struct ArrayAppendFromMake;
74
75template <typename T, typename R, typename... Args, R(*MakeFunc)(Args...)>
76struct ArrayAppendFromMake<T, MakeFunc>
77{
78 // Mirrors T::make signature for append IntelliSense without changing behavior.
79 T append(Args... args)
80 {
81 return static_cast<Array<T>&>(*this).template appendImpl<T>(std::forward<Args>(args)...);
82 }
83};

85template <typename Derived, typename T>
86struct ArrayAppendBase
87{
92 template <typename U = T, typename... Args,
93 std::enable_if_t<std::is_base_of_v<Base, U>, int> = 0>
94 U append(Args&&... args)
95 {
96 return static_cast<Derived&>(*this).template appendImpl<U>(std::forward<Args>(args)...);
97 }
98};

100template <typename Derived, typename T, typename = void>
101struct ArrayAppendOverloads : ArrayAppendBase<Derived, T> {};

103template <typename Derived, typename T>
104struct ArrayAppendOverloads<Derived, T, std::void_t<decltype(&T::make)>>
105 : ArrayAppendBase<Derived, T>,
106 ArrayAppendFromMake<T, &T::make>
107{
108 using ArrayAppendBase<Derived, T>::append;
109 using ArrayAppendFromMake<T, &T::make>::append;
110};

157class Base
158{
159public:
160 virtual ~Base() = default;
161
163 Base(const Base& src) : m_root(src.m_root), m_pointer(src.m_pointer)
164 {}
165
167 Base(Base&& src) noexcept : m_root(src.m_root), // m_root must be copied (not moved)
168 m_pointer(std::move(src.m_pointer))
169 {}
170
172 Base& operator=(const Base& src)
173 {
174 if (this != &src) {
175 if (m_root != src.m_root) {
176 throw std::logic_error("Assignment from a different JSON document is not allowed.");
177 }
178 m_pointer = src.m_pointer;
179 }
180 return *this;
181 }
182
184 Base& operator=(Base&& src)
185 {
186 if (this != &src) {
187 if (m_root != src.m_root) {
188 throw std::logic_error("Assignment from a different JSON document is not allowed.");
189 }
190 m_pointer = std::move(src.m_pointer);
191 }
192 return *this;
193 }
194
197 [[nodiscard]] std::string dump(int indents = -1) const
198 {
199 return ref().dump(indents);
200 }
201
205 template <typename T>
206 [[nodiscard]] T parent() const
207 {
208 static_assert(std::is_base_of_v<Base, T>, "Template type mush be derived from Base.");
209 return T(m_root, m_pointer.parent_pointer());
210 }
211
216 template <typename T, typename Scope = scope::Default>
217 [[nodiscard]] std::optional<T> getEnclosingElement() const;
218
220 [[nodiscard]] json_pointer pointer() const { return m_pointer; }
221
223 [[nodiscard]] Document document() const;
224
225protected:
231 [[nodiscard]] json& ref() const { return resolve_pointer(); }
232
237 [[nodiscard]] json& ref() { return resolve_pointer(); }
238
240 [[nodiscard]] const std::shared_ptr<json>& root() const { return m_root; }
241
247 Base(const std::shared_ptr<json>& root, json_pointer pointer)
248 : m_root(root), m_pointer(std::move(pointer)) {}
249
256 Base(json&& jsonRef, Base& parent, std::string_view key)
257 : m_root(parent.m_root), m_pointer(parent.m_pointer / std::string(key))
258 {
259 (*m_root)[m_pointer] = std::move(jsonRef);
260 }
261
269 template <typename T>
270 [[nodiscard]] T getChild(std::string_view key) const
271 {
272 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
273
274 json_pointer childPointer = m_pointer / std::string(key);
275 if (!checkKeyIsValid<T>(childPointer)) {
276 throw std::out_of_range("Child node not found: " + std::string(key));
277 }
278
279 return T(m_root, childPointer);
280 }
281
289 template <typename T>
290 T setChild(std::string_view key, const T& value)
291 {
292 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
293
294 json_pointer childPointer = m_pointer / std::string(key);
295 (*m_root)[childPointer] = value.ref();
296 return T(m_root, childPointer);
297 }
298
306 template <typename T>
307 [[nodiscard]] std::optional<T> getOptionalChild(std::string_view key) const
308 {
309 static_assert(std::is_base_of_v<Base, T>, "template type must be derived from Base");
310
311 json_pointer childPointer = m_pointer / std::string(key);
312 if (!checkKeyIsValid<T>(childPointer)) {
313 return std::nullopt;
314 }
315
316 return T(m_root, childPointer);
317 }
318
319private:
327 template <typename T>
328 [[nodiscard]] bool checkKeyIsValid(const json_pointer& pointer) const
329 {
330 if (!(*m_root).contains(pointer)) {
331 return false;
332 }
333
334 const json& node = (*m_root).at(pointer);
335
336 if constexpr (std::is_base_of_v<Object, T>) {
337 if (!node.is_object()) {
338 throw std::runtime_error("Expected an object for: " + pointer.to_string());
339 }
340 } else if constexpr (std::is_base_of_v<Array<typename T::value_type>, T>) {
341 if (!node.is_array()) {
342 throw std::runtime_error("Expected an array for: " + pointer.to_string());
343 }
344 }
345
346 return true;
347 }
348
354 [[nodiscard]] json& resolve_pointer() const
355 {
356 return (*m_root).at(m_pointer); // Throws if invalid
357 }
358
359 const std::shared_ptr<json> m_root;
360 json_pointer m_pointer;
361
362 friend class validation::SemanticValidator;
363};

452 public:

583 template <typename U, typename... Args>
584 U appendImpl(Args&&... args)
585 {
586 static_assert(std::is_base_of_v<Base, U>, "Array::appendImpl requires a Base-derived element type.");
587 if constexpr (std::is_base_of_v<Object, U>) {
588 ref().push_back(json::object());
589 } else {
590 ref().push_back(json::array());
591 }
592 return U(*this, std::to_string(ref().size() - 1), std::forward<Args>(args)...);
593 }

759template <typename E, typename = std::enable_if_t<std::is_enum_v<E>>>
760struct EnumStringMapping
761{
762 static const std::unordered_map<std::string, E> stringToEnum();
763
765 static const std::unordered_map<E, std::string> enumToString()
766 {
767 static const std::unordered_map<E, std::string> reverseMap = []() {
768 std::unordered_map<E, std::string> result;
769 for (const auto& element : EnumStringMapping<E>::stringToEnum()) {
770 result.emplace(element.second, element.first);
771 }
772 return result;
773 }();
774 return reverseMap;
775 }

781template <typename T>
782class Dictionary : public Object,
783 public detail::DictionaryAppendOverloads<Dictionary<T>, T>
784{
785 static_assert(std::is_arithmetic_v<T> || std::is_same_v<T, std::string> ||
786 std::is_base_of_v<ArrayElementObject, T>, "Invalid MNX dictionary element type.");
787
788private:
789 template <typename DictionaryType, typename IteratorType>
790 struct iter
791 {
792 using value_type = std::pair<const std::string, T>;
793 using difference_type = std::ptrdiff_t;
794 using iterator_category = std::forward_iterator_tag;
795 using pointer = value_type*;
796 using reference = value_type&;
797
798 private:
799 DictionaryType* m_ptr {};
800 IteratorType m_it {};
801 mutable std::optional<value_type> m_pair; // cached key/value
802
803 void update_pair() const
804 {
805 m_pair.reset();
806 if (m_it != m_ptr->ref().end()) {
807 m_pair.emplace(m_it.key(), m_ptr->valueForKey(m_it.key()));
808 }
809 }
810
811 public:
812 iter(DictionaryType* ptr, IteratorType it)
813 : m_ptr(ptr), m_it(it)
814 {
815 update_pair();
816 }
817
818 [[nodiscard]] reference operator*() const { return *m_pair; }
819 [[nodiscard]] pointer operator->() const { return &*m_pair; }
820
821 iter& operator++() { ++m_it; update_pair(); return *this; }
822 iter operator++(int) { iter tmp(*this); ++(*this); return tmp; }
823
824 [[nodiscard]] bool operator!=(const iter& o) const { return m_it != o.m_it; }
825 [[nodiscard]] bool operator==(const iter& o) const { return m_it == o.m_it; }
826 };
827
828public:
830 using value_type = T;
831
832 using iterator = iter<Dictionary, json::iterator>;
833 using const_iterator = iter<const Dictionary, json::const_iterator>;
834
838 Dictionary(const std::shared_ptr<json>& root, json_pointer pointer)
839 : Object(root, pointer)
840 {
841 }
842
846 Dictionary(Base& parent, std::string_view key)
847 : Object(parent, key) {}
848
850 [[nodiscard]] size_t size() const { return ref().size(); }
851
853 [[nodiscard]] bool empty() const { return ref().empty(); }
854
856 void clear() { ref().clear(); }
857
861 [[nodiscard]] T at(std::string_view key) const
862 { return valueForKey(key); }
863
865 template <typename U = T>
866 std::enable_if_t<!std::is_base_of_v<Base, U>, void>

869 ref()[key] = value;

906

909

912
913private:
915 [[nodiscard]] T valueForKey(std::string_view key) const
916 {
917 if constexpr (std::is_base_of_v<Base, T>) {
918 return getChild<T>(key);
919 } else {
920 return getChild<SimpleType<T>>(key);
921 }
922 }

925 U appendImpl(std::string_view key, Args&&... args)
926 {
927 static_assert(std::is_base_of_v<Base, U>, "Dictionary::appendImpl requires a Base-derived element type.");
928 if constexpr (std::is_base_of_v<Object, U>) {
929 ref()[key] = json::object();
930 } else {
931 ref()[key] = json::array();
932 }
933 return U(*this, key, std::forward<Args>(args)...);
934 }

972 if (it == map.end()) {
974 j = BasicJsonType();
975 return;
976 }
977 j = it->second;
978 }
